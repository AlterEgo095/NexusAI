   107→/* ─── Helpers ─── */
   108→function relativeTime(date: Date | null): string {
   109→  if (!date) return 'Jamais'
   110→  const now = new Date()
   111→  const diff = now.getTime() - date.getTime()
   112→  const minutes = Math.floor(diff / 60000)
   113→  const hours = Math.floor(minutes / 60)
   114→  const days = Math.floor(hours / 24)
   115→  if (minutes < 1) return "À l'instant"
   116→  if (minutes < 60) return `Il y a ${minutes} min`
   117→  if (hours < 24) return `Il y a ${hours}h`
   118→  if (days < 7) return `Il y a ${days}j`
   119→  return date.toLocaleDateString('fr-FR', { day: 'numeric', month: 'short' })
   120→}

   450→/* ─── Workflow Editor ─── */
   451→function WorkflowEditor({
   452→  workflow,
   453→  onUpdate,
   454→  onBack,
   455→}: {
   456→  workflow: WorkflowType
   457→  onUpdate: (updates: Partial<WorkflowType>) => void
   458→  onBack: () => void
   459→}) {
   460→  const [selectedNodeId, setSelectedNodeId] = useState<string | null>(null)
   461→  const [dragInfo, setDragInfo] = useState<{ nodeId: string; offsetX: number; offsetY: number } | null>(null)
   462→  const canvasRef = useRef<HTMLDivElement>(null)
   463→  const [addMenuOpen, setAddMenuOpen] = useState(false)
   464→
   465→  const selectedNode = workflow.nodes.find((n) => n.id === selectedNodeId) || null
   466→
   467→  /* ─── Drag handlers ─── */
   468→  const handlePointerDown = useCallback(
   469→    (e: React.PointerEvent, nodeId: string) => {
   470→      e.preventDefault()
   471→      const node = workflow.nodes.find((n) => n.id === nodeId)
   472→      if (!node) return
   473→      setDragInfo({
   474→        nodeId,
   475→        offsetX: e.clientX - node.position.x,
   476→        offsetY: e.clientY - node.position.y,
   477→      })
   478→      setSelectedNodeId(nodeId)
   479→      ;(e.target as HTMLElement).setPointerCapture(e.pointerId)
   480→    },
   481→    [workflow.nodes]
   482→  )
   483→
   484→  const handlePointerMove = useCallback(
   485→    (e: React.PointerEvent) => {
   486→      if (!dragInfo) return
   487→      const canvas = canvasRef.current
   488→      if (!canvas) return
   489→      const rect = canvas.getBoundingClientRect()
   490→      const x = Math.max(0, e.clientX - rect.left - dragInfo.offsetX + canvas.scrollLeft)
   491→      const y = Math.max(0, e.clientY - rect.top - dragInfo.offsetY + canvas.scrollTop)
   492→      const updatedNodes = workflow.nodes.map((n) =>
   493→        n.id === dragInfo.nodeId ? { ...n, position: { x: Math.round(x), y: Math.round(y) } } : n
   494→      )
   495→      onUpdate({ nodes: updatedNodes })
   496→    },
   497→    [dragInfo, workflow.nodes, onUpdate]
   498→  )
   499→
   500→  const handlePointerUp = useCallback(() => {
   501→    setDragInfo(null)
   502→  }, [])
   503→
   504→  /* ─── Node config handlers ─── */
   505→  const updateNodeConfig = useCallback(
   506→    (nodeId: string, updates: Partial<WorkflowNode>) => {
   507→      const updatedNodes = workflow.nodes.map((n) =>
   508→        n.id === nodeId ? { ...n, ...updates } : n
   509→      )
   510→      onUpdate({ nodes: updatedNodes })
   511→    },
   512→    [workflow.nodes, onUpdate]
   513→  )
   514→
   515→  const addNode = useCallback(
   516→    (type: WorkflowNode['type']) => {
   517→      const newNode: WorkflowNode = {
   518→        id: `node-${Date.now()}`,
   519→        type,
   520→        label: type === 'trigger' ? 'Nouveau Déclencheur' : type === 'action' ? 'Nouvelle Action' : type === 'condition' ? 'Nouvelle Condition' : 'Nouvelle Sortie',
   521→        config: {},
   522→        position: { x: 200 + Math.random() * 300, y: 100 + Math.random() * 200 },
   523→      }
   524→      onUpdate({ nodes: [...workflow.nodes, newNode] })
   525→      setSelectedNodeId(newNode.id)
   526→      setAddMenuOpen(false)
   527→    },
   528→    [workflow.nodes, onUpdate]
   529→  )
   530→
   531→  const deleteNode = useCallback(
   532→    (nodeId: string) => {
   533→      const updatedNodes = workflow.nodes.filter((n) => n.id !== nodeId)
   534→      onUpdate({ nodes: updatedNodes })
   535→      if (selectedNodeId === nodeId) setSelectedNodeId(null)
   536→    },
